SELECT 
    pnk.ma_phieu_nhap,
    pnk.ngay_nhap_hang,
    dt.ten_doi_tac,
    k.ten_kho,
    pnk.nguoi_giao_hang,
    pnk.nguoi_tao,
    vt.ma_vat_tu,
    vt.ten_vat_tu,
    ms.ten_mau,
    ctnk.so_luong_thuc_nhan AS so_luong,
    dvt.ten_dvt,
    pnk.ghi_chu AS ghi_chu_phieu,
    ctnk.ghi_chu AS ghi_chu_chi_tiet
FROM public.tr_phieu_nhap_kho pnk
LEFT JOIN public.md_doi_tac dt ON pnk.id_doi_tac = dt.id
LEFT JOIN public.md_dinh_danh_kho k ON pnk.id_kho = k.id
JOIN public.tr_chi_tiet_nhap_kho ctnk ON pnk.id = ctnk.id_phieu_nhap
JOIN public.md_danh_muc_vat_tu vt ON ctnk.id_vat_tu = vt.id
LEFT JOIN public.md_don_vi_tinh dvt ON vt.id_don_vi_tinh = dvt.id
LEFT JOIN public.md_mau_sac ms ON ctnk.id_mau_sac = ms.id
WHERE 
    DATE(pnk.ngay_nhap_hang) >= {{moment(dtp_tu_ngay.selectedDate).format('YYYY-MM-DD')}}::date
    AND DATE(pnk.ngay_nhap_hang) <= {{moment(dtp_den_ngay.selectedDate).format('YYYY-MM-DD')}}::date
    AND ({{sel_ma_phieu_nhap_tracuu.selectedOptionValue || ''}} = '' OR pnk.ma_phieu_nhap = {{sel_ma_phieu_nhap_tracuu.selectedOptionValue}})
    AND ({{sel_don_vi_giao_hang_tracuu.selectedOptionValue || ''}} = '' OR pnk.id_doi_tac::text = {{sel_don_vi_giao_hang_tracuu.selectedOptionValue}})
ORDER BY pnk.ngay_nhap_hang DESC, pnk.ma_phieu_nhap DESC;
