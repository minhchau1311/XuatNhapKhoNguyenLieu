SELECT 
    dm.ma_don_mua,
    dt.ten_doi_tac,
    dm.created_at,
    dm.ghi_chu,
    (SELECT count(*) FROM public.tr_chi_tiet_don_mua WHERE id_don_mua = dm.id) as tong_so_mon
FROM public.tr_don_mua_hang dm
JOIN public.md_doi_tac dt ON dm.id_doi_tac = dt.id
WHERE dm.id::text = '{{sel_po_nguon.selectedOptionValue}}';
