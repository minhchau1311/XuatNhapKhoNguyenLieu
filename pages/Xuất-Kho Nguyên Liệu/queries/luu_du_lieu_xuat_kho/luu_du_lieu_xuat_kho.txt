WITH new_phieu_xuat AS (
    INSERT INTO public.tr_phieu_xuat_kho (
        ma_phieu_xuat, id_kho_xuat, id_xuong_nhan, nguoi_nhan_hang, ghi_chu, nguoi_tao
    ) VALUES (
        {{inp_ma_phieu_xuat.text}},
        {{sel_kho_xuat.selectedOptionValue}}::uuid,
        {{sel_xuong_nhan.selectedOptionValue}}::uuid,
        {{inp_nguoi_nhan.text}},
        {{inp_ghi_chu_vo.text}},
        {{appsmith.user.email}} 
    ) RETURNING id
),
chi_tiet_xuat AS (
    INSERT INTO public.tr_chi_tiet_xuat_kho (
        id_phieu_xuat, id_vat_tu, id_mau_sac, so_luong_thuc_xuat, ly_do_xuat, ghi_chu, nguoi_tao
    )
    SELECT 
        (SELECT id FROM new_phieu_xuat),
        (elem->>'id_vat_tu')::uuid,
        NULLIF(elem->>'id_mau_sac', '')::uuid,
        (elem->>'so_luong')::numeric,
        elem->>'ly_do_xuat', 
        elem->>'ghi_chu',
        {{appsmith.user.email}}
    FROM jsonb_array_elements({{JSON.stringify(appsmith.store.bangTamXK)}}::jsonb) AS elem
    RETURNING id, id_vat_tu, id_mau_sac, so_luong_thuc_xuat
)
INSERT INTO public.tr_nhat_ky_kho (
    loai_giao_dich, id_kho, id_vat_tu, id_mau_sac, so_luong, nguoi_thuc_hien, nguoi_tao, id_chi_tiet_xuat_kho
)
SELECT 
    'XUAT_KHO',
    {{sel_kho_xuat.selectedOptionValue}}::uuid,
    id_vat_tu,
    id_mau_sac,
    -(so_luong_thuc_xuat), 
    {{appsmith.user.name || appsmith.user.email}}, 
    {{appsmith.user.email}},                       
    id
FROM chi_tiet_xuat
RETURNING id;
