SELECT
    px.ma_phieu_xuat AS "Mã Phiếu Xuất",
    TO_CHAR(px.created_at, 'DD/MM/YYYY HH24:MI') AS "Ngày Xuất",
    bp.ten_bo_phan AS "Đơn Vị Nhận",
    dh.ma_don_hang AS "Mã Đơn Hàng", -- Bây giờ lấy sẽ có dữ liệu
    btp.ma_vat_tu AS "Mã BTP",
    vt.ma_vat_tu AS "Mã Vật Tư",
    vt.ten_vat_tu AS "Tên Vật Tư",
    ms.ten_mau AS "Màu Sắc",
    ctx.so_luong_thuc_xuat AS "Số Lượng",
    dvt.ten_dvt AS "ĐVT",
    ctx.ly_do_xuat AS "Lý Do Xuất",
    px.nguoi_tao AS "Người Tạo",
    px.nguoi_nhan_hang AS "Người Nhận",
    px.ghi_chu AS "Ghi Chú"
FROM public.tr_phieu_xuat_kho px
LEFT JOIN public.md_bo_phan bp ON px.id_xuong_nhan = bp.id
JOIN public.tr_chi_tiet_xuat_kho ctx ON px.id = ctx.id_phieu_xuat
JOIN public.md_danh_muc_vat_tu vt ON ctx.id_vat_tu = vt.id
LEFT JOIN public.md_don_vi_tinh dvt ON vt.id_don_vi_tinh = dvt.id
LEFT JOIN public.md_mau_sac ms ON ctx.id_mau_sac = ms.id
LEFT JOIN public.md_danh_muc_vat_tu btp ON ctx.id_btp = btp.id

-- ĐÃ FIX: Nối thẳng từ Vỏ Phiếu Xuất (px) sang Đơn Hàng (dh)
LEFT JOIN public.tr_thong_tin_don_hang dh ON ctx.id_don_hang = dh.id

WHERE
    DATE(px.created_at) >= {{ dtp_tu_ngay_xuat.selectedDate ? moment(dtp_tu_ngay_xuat.selectedDate).format('YYYY-MM-DD') : '2000-01-01' }}::date
    AND DATE(px.created_at) <= {{ dtp_den_ngay_xuat.selectedDate ? moment(dtp_den_ngay_xuat.selectedDate).format('YYYY-MM-DD') : '2100-01-01' }}::date
    AND ({{sel_ma_phieu_xuat.selectedOptionValue || ''}} = '' OR px.ma_phieu_xuat = {{sel_ma_phieu_xuat.selectedOptionValue}})
    AND ({{sel_bo_phan_nhan.selectedOptionValue || ''}} = '' OR px.id_xuong_nhan::text = {{sel_bo_phan_nhan.selectedOptionValue}})
ORDER BY px.created_at DESC, px.ma_phieu_xuat DESC;
